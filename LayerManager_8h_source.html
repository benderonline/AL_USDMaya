179  // attributes to store the serialised layers (used for file IO only)
180 
181  // Note that the layers attribute should ONLY used during serialization, as this is the ONLY
182  // times at which these attributes are guaranteed to "line up" to the internal layer register
183  // (m_layerList). Ie, immediately before save (due to the pre-save callback), the attributes will
184  // be written from m_layerList; and immediate after open (due to the post-open callback), m_layerList
185  // will be initialized from the attributes. At all other times, the attributes will be OUT OF SYNC
186  // (and, in fact, are intentionally set to be "empty", so there's no confusion / someone doesn't
187  // try to use "out of date" information)
188  AL_DECL_ATTRIBUTE(layers);
189  // Not using AL_DECL_ATTRIBUTE for these, because we never want a generic, ie, identifierPlug() -
190  // they only make sense for a particular index of the parent array-attribute... and it taking up
191  // the "identifierPlug" name is confusing
192  AL_DECL_MULTI_CHILD_ATTRIBUTE(identifier);
193  AL_DECL_MULTI_CHILD_ATTRIBUTE(serialized);
194  AL_DECL_MULTI_CHILD_ATTRIBUTE(anonymous);
